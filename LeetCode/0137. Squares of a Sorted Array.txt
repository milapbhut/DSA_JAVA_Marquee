class Solution {
public int[] sortedSquares(int[] nums) {
for(int i = 0; i < nums.length; i++){
int temp = nums[i] * nums[i];
nums[i] = temp;
}
for(int i = 0; i < nums.length; i++){
for(int j = i + 1; j < nums.length; j++){
}
if(nums[i] > nums[j]){
}
int temp = nums[j];
}
}
nums[j] = nums[i];
nums[i] = temp;
return nums;
}
